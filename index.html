// Toggle Menu function toggleMenu() { const menu = document.getElementById("menu"); if (menu.style.display === "block") { menu.style.display = "none"; } else { menu.style.display = "block"; } } // Function to Open Blog Page function openBlog(filename) { window.location.href = filename; } // Attach Event Listeners to Blog Links document.addEventListener("DOMContentLoaded", function () { const blogLinks = document.querySelectorAll(".blog-link"); blogLinks.forEach(link => { link.addEventListener("click", function (event) { event.preventDefault(); const blogFile = this.getAttribute("data-file"); openBlog(blogFile); }); }); });
